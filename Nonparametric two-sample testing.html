Introduction

The nonparametric two-sample hypothesis testing problem involves: given X_1, X_2, …, X_n i.i.d. F and Y_1, Y_2, …, Y_m i.i.d. G, test the null hypothesis of F = G against the alternative hypothesis of F ≠≠ G. The test statistic is based on embedding F and G into a reproducing kernel Hilbert space and then compute a distance between the resulting embeddings. For this primitive, the Hilbert space is associated with the Gaussian kernel.

# Generate Adjacency Matrix
n <- 100
blocks <- c(rep(0, n*0.4), rep(1, n*0.6))
A <- matrix(rep(0, n*n), nrow = n)
for (i in 2:n) {
  for (j in 1:(i-1)) {
    if (blocks[i] == blocks[j]) {
      A[i, j] <- rbinom(1, 1, 0.5)
    } else {
      A[i, j] <- rbinom(1, 1, 0.2)
    }
    A[j, i] <- A[i, j]
  }
}

# Generate Adjacency Matrix with epsilon = 0.02
n <- 100
epsilon <- 0.02
blocks <- c(rep(0, n*0.4), rep(1, n*0.6))
Ahat <- matrix(rep(0, n*n), nrow = n)
for (i in 2:n) {
  for (j in 1:(i-1)) {
    if (blocks[i] == blocks[j]) {
      Ahat[i, j] <- rbinom(1, 1, 0.5 + epsilon)
    } else {
      Ahat[i, j] <- rbinom(1, 1, 0.2)
    }
    Ahat[j, i] <- Ahat[i, j]
  }
}

# Embed to R^2
SVD_X = svd(A)
X = SVD_X$u %*% (diag(SVD_X$d) ^ 1/2)
SVD_Y = svd(Ahat)
Y = SVD_Y$u %*% (diag(SVD_Y$d) ^ 1/2)
# Xhat = SVD_X$u[,1:2] %*% (diag(SVD_X$d[1:2]^ 1/2))
Xhat = graphstats::ase(A,2)
# Yhat = SVD_Y$u[,1:2] %*% (diag(SVD_Y$d[1:2]^ 1/2))
Yhat = graphstats::ase(Ahat,2)
test_stats_1 = graphstats::nonpar(X,Y,sigma = 0.5)
test_stats_2 = graphstats::nonpar(Xhat,Yhat,sigma = 0.5)
print(test_stats_1)
